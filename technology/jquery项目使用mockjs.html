  • jquery项目中使用mockjs

    • 在项目目录下添加mock文件夹并在里面创建mock.js
      • 在html页面中引入jquery和mockjs

      # 在项目目录下添加mock文件夹并在里面创建mock.js

      Mock.mock("/user/userinfo", "get", {
    id: "@id",
    name: "@cname",
    date: "@date()",
    avatar: "@image('200x100', '#50B347', '#FFF', 'Mock.js')",
    ip: "@ip()",
    email: "@email",
});

      # 在html页面中引入jquery和mockjs

      <!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>Document</title>
        <script src="https://cdn.bootcss.com/jquery/2.2.4/jquery.min.js"></script>
        <script src="https://cdn.bootcss.com/Mock.js/1.0.0/mock-min.js"></script>
        <script src="./mock/mock.js"></script>
    </head>
    <body>
        <script>
            $.ajax({
                url: "/user/userinfo",
                //将返回的数据解析成json
                dataType: "json",
                success(data) {
                    console.log(data);
                },
            });
        </script>
    </body>
</html>
